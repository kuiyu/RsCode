定义数据结构

csharp
[StructLayout(LayoutKind.Sequential, Pack = 1)]
public struct SensorData : unmanaged
{
    public long Timestamp;
    public double Temperature;
    public double Humidity;
    
    public override string ToString() => 
        $"[{Timestamp}] Temp: {Temperature:F1}C, Humid: {Humidity:F1}%";
}

csharp
public MainForm()
{
    InitializeComponent();
    //启动接收线程消息
		new Thread(ReceiveMessages) { IsBackground = true }.Start();
}

void ReceiveMessages()
{
    try
    {
        // 消费者代码 
        using (var helper = MemoryMappedFileHelper.CreateOrOpen("SensorStream", 1024))
        {
            while (true)
            {
				var data = helper.Read<SensorData>(0);
                 if(data!=null)
                {
                    //更新UI
                    Invoke((MethodInvoker)delegate
                    {
                        taskViewList.refreshStatus(data);
                    });
                }
                else
                {
                    //Console.WriteLine("等待数据超时");
                } 
            }
        }
    }
    catch (Exception ex)
    {
        LogHelper.Info(ex.Message);
    }
}
